```elm
module Example exposing (Article, articleDecoder)

import Json.Decode as Decode
import Json.Decode.Pipeline as DecodePipeline

type alias Article =
    { title : String
    , subtext : String
    , tags : List String
    , thumbnail : Maybe String
    }

articleDecoder : Decoder Article
articleDecoder =
    DecodePipeline.decode Article
        |> DecodePipeline.required "title" Decode.string
        |> DecodePipeline.required "subtext" Decode.string
        |> DecodePipeline.required "tags" (Decode.list Decode.string)
        |> DecodePipeline.optional "thumbnail" Decode.string Nothing
```
